  • 访问数组，方括号里边是整数

// 虽然是一个数组对象，但是不能用正常的创建方法

int[] myIntArray = new int[]{1, 2, 3 ,4 ,5};
// 需要注意的是，这两个数字必须相同

new

创建实例，调用实例构造器/初始化器，并返回实例的内存地址。

构造器是圆括号，初始化器是花括号。

// 关键字var来自variable。
var x = 100;
x.GetType().Name

// 以下语句可以正常调用，考虑算符优先级
new Form().ShowDialog();
// 但是在内存中创建对象却并不一定需要new。例如以下语法糖
string name = "Tom";
int[] myArray = {1,2,3,4};
// 创建匿名对象
var person = new {Name = "tom", age = 34};
//<>f__AnonymousType0`2
// 0代表是第0个，2代表里边有两个东西
person.GetType().Name

new 还可以做为关键字。

  • 重写子类的方法

checked/unchecked

检查值在内存中是否有溢出

  uint x = uint.MaxValue;
  Console.WriteLine(x);
  string binStr = Convert.ToString(x, 2);
  Console.WriteLine(binStr);
  try {
    // 如果不检查异常，就不会有异常，y就是0
  	uint y = unchecked(x + 1);
    // 如果有异常，就会检查异常，就会触发下面的语句，如果没有catch，会报错
  	Console.WriteLine(y);
  }
  catch (OverflowException ex) {
  	Console.WriteLine("OverFlow sized");
  }

delegate

把delegate作为操作符使用已经很少了，取代它的是lambda。都是用于匿名方法。

this.myButton.click	+= delegate(object sender, RoutedEventArgs e){
	this.MyTextBox.Text = "Hello World!";
}

this.myButton.click	+= (sender, e) => {
	this.MyTextBox.Text = "Hello World!";
}

自己定义一个类型转换

// 有一个类，它叫Stone，还有一个类，它叫Monkey。地球人都知道这个事情，编译器也知道。
// 传闻太古年间有一则秘闻，其实，石头机缘巧合（cast）之下可以变成猴子（对！就是西游记）
// 但是还有人不知道这件事。编译器也不知道。所以！
using System;
using System.Collections.Generic;

namespace TieMenLiu {
    class TieMen{
        static void Main(string[] args) {
            Stone stone = new Stone();
            stone.Age = 5000;
            /*Monkey wukongsun = (Monkey)stone;*/
            Monkey wukongsun = stone;
            Console.WriteLine(wukongsun.Age);
        }        
    }
    public class Stone {
        public int Age;

        // 不能同时显示和隐式类型转换
        /*public static explicit operator Monkey(Stone stone) {
            Monkey m = new Monkey();
            m.Age = stone.Age / 500;
            return m;
        }*/

        public static implicit operator Monkey(Stone stone) {
            Monkey m = new Monkey();
            m.Age = stone.Age / 250;
            return m;
        }
    }

    public class Monkey {
        public int Age;
    }
}
